
DROP TABLE IF EXISTS public.tbl_tax_calc_part_one;

CREATE TABLE IF NOT EXISTS public.tbl_tax_calc_part_one
(
    tax_calc_part_one_id bigint NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 9223372036854775807 CACHE 1 ),
   
    province text  ,    
	regional_assistance_name text  , 
	region_fid bigint , 
	labour_or_expense text  , 
	credit_percentage decimal(10,2) ,
	max_credit_allowed bigint ,    
    created_at timestamp with time zone DEFAULT 'now()',
    updated_at timestamp with time zone,
    CONSTRAINT tbl_tax_calc_part_one_id PRIMARY KEY (tax_calc_part_one_id),
    CONSTRAINT tbl_region_tax_calc_one_fid_fkey FOREIGN KEY (region_fid)
        REFERENCES public.tbl_region (region_id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE NO ACTION
);
